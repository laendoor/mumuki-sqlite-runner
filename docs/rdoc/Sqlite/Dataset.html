def initialize(data)
  @header = @rows = []
  rows = split_lines data
  unless rows.empty?
    @header = split_pipe rows.shift
    @rows = rows.map { |item| split_pipe item }
  end
end

def compare(other)
  same_header = same_header other.header
  same_rows   = same_rows other.rows

  if same_header & same_rows
    :equals
  elsif !same_header
    :distinct_columns
  else
    :distinct_rows
  end
end
