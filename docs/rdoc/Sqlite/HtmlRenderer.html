def diff_class_of(value)
  case value
    when '+'
      'required'
    when '-'
      'errored'
    else
      'nothing'
  end
end

def parse_dataset(header, rows)
  header_sign = header.shift
  rows = rows.map do |row|
    {
        sign: row.shift,
        row: row
    }
  end

  {
      header: {
          sign: header_sign,
          class: diff_class_of(header_sign),
          fields: header
      },
      rows: rows.map do |row|
        {
            sign:row[:sign],
            class: diff_class_of(row[:sign]),
            fields: row[:row]
        }
      end
  }

end
